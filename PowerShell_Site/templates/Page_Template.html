Test if using Powershell Version 5.1

function Test-LatestPSVersion {
     $Version = '{0}.{1}' -f ( $PSVersionTable.PSVersion.Major ), ( $PSVersionTable.PSVersion.Minor )
     if ($Version -eq  '5.1') {
         return $true
     } else {
         return $false
     }
}
